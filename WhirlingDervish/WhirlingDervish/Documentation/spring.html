Spring

Summary

    The Spring class represents a specific type of spring for simulation.
It primarily holds data that characterizes a spring, with basic methods to
manipulate those data members.  The constructor can be a default constructor or
take values for most of the data members.

Data member description

innate length         This values represents the rest length of the spring
ks                    spring constant
kd                    spring dampner constant
fluctuation delta     percent of initial innate length by which the
                      innate length can vary
fluctuation start     percent of fluctuation delta to represent the actual
                      starting innate length of the spring

Header

class Spring {
public:
  Spring(double innateLength    =SpringConstraints::DEFAULT_INNATE_LENGTH,
	 double ks              =SpringConstraints::DEFAULT_KS,
	 double kd              =SpringConstraints::DEFAULT_KD,
	 double fluctuationDelta=SpringConstraints::DEFAULT_FLUCTUATION_DELTA,
	 double fluctuationStart=SpringConstraints::DEFAULT_FLUCTUATION_START);
  Spring(const Spring &);
  ~Spring(void);

  Spring &operator=(const Spring &);

  double  InnateLength(void)             const { return d_innateLength; }
  double &InnateLength(void)                   { return d_innateLength; }
  double  ks(void)                       const { return d_ks; }
  double &ks(void)                             { return d_ks; }
  double  kd(void)                       const { return d_kd; }
  double &kd(void)                             { return d_kd; }
  double  FluctuationDelta(void)         const { return d_fluctuationDelta; }
  double &FluctuationDelta(void)               { return d_fluctuationDelta; }
  double  FluctuationStart(void)         const { return d_fluctuationStart; }
  double &FluctuationStart(void)               { return d_fluctuationStart; }

  void ChangeKS(double delta) { d_ks += delta; return; }
  void ChangeKD(double delta) { d_kd += delta; return; }
  void ChangeFluctuationDelta(double delta) {d_fluctuationDelta+=delta;return;}
  void ChangeFluctuationStart(double delta) {d_fluctuationStart+=delta;return;}

  void Flucuate(void);

private:
  double d_innateLength;
  double d_ks;
  double d_kd;
  double d_fluctuationDelta;
  double d_fluctuationStart;
  double d_increment;
  double d_incrementDelta;
};
